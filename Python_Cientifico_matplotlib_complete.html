pl.ylabel('calculado')

Varias lineas en un mismo plot...

In [3]:
x = arange(100)
linear = arange(100)
square = [v * v for v in arange(0, 10, 0.1)]
In [4]:
lineas = pl.plot(x, linear, x, square)
leyenda = pl.legend(('linear', 'square'))

Ajustamos a gusto y piacere...

In [5]:
lineas = plot(x, linear, 'g+', x, square, 'ro')
leyenda = legend(('linear', 'square'), loc='upper left')

Figuras, plots y subplots...

In [6]:
def f(t):
    return np.exp(-t) * np.cos(2*np.pi*t)

t1 = np.arange(0.0, 5.0, 0.1)
t2 = np.arange(0.0, 5.0, 0.02)

plt.figure(1)
plt.subplot(211)
plt.plot(t1, f(t1), 'bo', t2, f(t2), 'k')

plt.subplot(212)
plt.plot(t2, np.cos(2*np.pi*t2), 'r--')
Out [6]:
[<matplotlib.lines.Line2D at 0xa1388ec>]
